#+BEGIN_SRC sh
  git clone https://github.com/Jeanhwea/rust-leetcode-cli.git
  cd rust-leetcode-cli
  cargo install --path .
#+END_SRC

  # ~/.leetcode/leetcode.toml

  [code]

  start_marker = "BEG_SUBMIT"
  end_marker = "END_SUBMIT"
  comment_problem_desc = true
  comment_leading = "//"
  test = false

  csrf = '...'
  session = 'ey...'

  [storage]
  cache = 'Problems'

  scripts = 'scripts'
